Install $ npm install --save ic-datepicker  (or) $ yarn add ic-datepicker
Import 
import { IcDatepickerModule } from 'ic-datepicker';

@NgModule({
  imports: [
    ...
    IcDatepickerModule,
  ],
})
class AppModule { }
Style 
// Angular CLI
"styles": [
  "../node_modules/ic-datepicker/themes/theme-default.css"
],

// Other
<link src="node_modules/ic-datepicker/themes/theme-default.css" rel="stylesheet">
Use 
<ic-datepicker options="datepickerOptions"></ic-datepicker>
